CRUD: Create & Read

Create (INSERT)

function createEmployee($db, $input) { $stmt = $db->prepare( "INSERT INTO employees (name, email, position, salary) VALUES (:name, :email, :position, :salary)" ); $stmt->execute($input); return $db->lastInsertId(); }

Read (SELECT)

function getAllEmployees($db) { $stmt = $db->query("SELECT * FROM employees"); return $stmt->fetchAll(); }

CRUD: Update & Delete

Update

function updateEmployee($db, $id, $input) { $stmt = $db->prepare( "UPDATE employees SET name=:name, email=:email, salary=:salary WHERE id=:id" ); $input['id'] = $id; $stmt->execute($input); }

Delete

function deleteEmployee($db, $id) { $stmt = $db->prepare("DELETE FROM employees WHERE id=:id"); $stmt->execute(['id' => $id]); }
